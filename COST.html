    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getFirestore, collection, addDoc, query, onSnapshot, serverTimestamp, orderBy, doc, deleteDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        import { getAuth, signInAnonymously, signOut, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";


        const firebaseConfig = {

            apiKey: "AIzaSyA57M4Y_qXNA60BKP5GycyJnlAZEmcRteI",

            authDomain: "ppkphb-ledger.firebaseapp.com",

            projectId: "ppkphb-ledger",

            storageBucket: "ppkphb-ledger.firebasestorage.app",

            messagingSenderId: "636544970929",

            appId: "1:636544970929:web:e5b8385cc0fb0978c1b290"

        };

        

        const finalAppId = "ppkphb-production";

        const app = initializeApp(firebaseConfig);

        const auth = getAuth(app);

        const db = getFirestore(app);

        let user, currentGroupId, currentUserRole, unsubscribe = null, pendingDeleteId = null;

        let selectedCategory = '食';


        // 輸入金額格式化

        window.formatInputAmount = (input) => {

            let value = input.value.replace(/[^0-9]/g, '');

            input.value = value !== '' ? Number(value).toLocaleString() : '';

        };


        // 點選類別功能

        window.setCategory = (cat) => {

            selectedCategory = cat;

            document.querySelectorAll('.cat-btn').forEach(btn => {

                btn.classList.toggle('btn-active', btn.getAttribute('data-cat') === cat);

            });

        };


        window.selectUser = (role) => {

            currentUserRole = role;

            document.getElementById('btn-user-ppk').className = role === 'PPK' ? 'btn-outline py-2 rounded font-bold uppercase btn-active' : 'btn-outline py-2 rounded font-bold uppercase';

            document.getElementById('btn-user-phb').className = role === 'PHB' ? 'btn-outline py-2 rounded font-bold uppercase btn-active' : 'btn-outline py-2 rounded font-bold uppercase';

        };


        window.startApp = async () => {

            const gid = document.getElementById('group-id').value.trim();

            if (!gid || !currentUserRole) return alert("請完整填寫代碼並選擇身分");

            currentGroupId = gid;

            try { await signInAnonymously(auth); } catch (e) { console.error(e); }

        };


        onAuthStateChanged(auth, (u) => {

            if (u) {

                user = u;

                document.getElementById('display-group').innerText = currentGroupId;

                document.getElementById('display-role').innerText = currentUserRole;

                document.getElementById('setup-screen').classList.add('hidden');

                document.getElementById('main-app').classList.remove('hidden');

                listenToData();

            } else {

                document.getElementById('setup-screen').classList.remove('hidden');

                document.getElementById('main-app').classList.add('hidden');

            }

        });


        window.switchUser = async () => { if (unsubscribe) unsubscribe(); currentUserRole = null; currentGroupId = null; await signOut(auth); };


        function listenToData() {

            const q = collection(db, 'artifacts', finalAppId, 'public', 'data', 'expenses');

            unsubscribe = onSnapshot(q, (snapshot) => {

                const records = [];

                snapshot.forEach((doc) => {

                    const data = doc.data();

                    if (data.groupId === currentGroupId) records.push({ id: doc.id, ...data });

                });

                

                records.sort((a, b) => {

                    const timeA = a.createdAt ? a.createdAt.toDate().getTime() : 0;

                    const timeB = b.createdAt ? b.createdAt.toDate().getTime() : 0;

                    return timeB - timeA;

                });


                renderUI(records);

            });

        }


        function renderUI(records) {

            const list = document.getElementById('list-container');

            let tA = 0, tB = 0, tM = 0;

            let cats = { '食': 0, '衣': 0, '住': 0, '行': 0, '樂': 0 };

            const now = new Date();

            const todayStr = now.toLocaleDateString();


            list.innerHTML = records.length ? '' : '<div class="col-span-full text-center py-20 text-gray-300 italic">目前尚無紀錄</div>';

            

            records.forEach(rec => {

                const amt = Number(rec.amount) || 0;

                const d = rec.createdAt ? rec.createdAt.toDate() : new Date();

                

                if (d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear()) {

                    tM += amt;

                    if (cats.hasOwnProperty(rec.category)) cats[rec.category] += amt;

                }

                

                if (d.toLocaleDateString() === todayStr) {

                    if (rec.userRole === 'PPK') tA += amt; else tB += amt;

                }


                const card = document.createElement('div');

                card.className = 'flex items-center justify-between p-3 bw-border rounded bg-white animate-fade-in shadow-sm hover:shadow-md transition-shadow';

                card.innerHTML = `

                    <div class="flex items-center gap-3 overflow-hidden">

                        <div class="flex-shrink-0 w-8 h-8 rounded-full bg-black text-white flex items-center justify-center font-bold text-xs">${rec.category}</div>

                        <div class="truncate">

                            <div class="font-bold text-sm truncate leading-tight">${rec.name}</div>

                            <div class="text-[9px] text-gray-400 font-bold uppercase tracking-tighter">${rec.userRole} • ${d.toLocaleDateString()}</div>

                        </div>

                    </div>

                    <div class="flex items-center gap-2 flex-shrink-0 pl-2">

                        <div class="text-lg font-black italic">$${amt.toLocaleString()}</div>

                        <button class="del-btn text-gray-200 hover:text-red-600 transition-colors">

                            <svg xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5" stroke-linecap="round" stroke-linejoin="round"><path d="M3 6h18"/><path d="M19 6v14c0 1-1 2-2 2H7c-1 0-2-1-2-2V6"/><path d="M8 6V4c0-1 1-2 2-2h4c1 0 2 1 2 2v2"/></svg>

                        </button>

                    </div>`;

                card.querySelector('.del-btn').onclick = () => { pendingDeleteId = rec.id; document.getElementById('delete-modal').classList.remove('hidden'); };

                list.appendChild(card);

            });


            document.getElementById('total-a').innerText = `$ ${tA.toLocaleString()}`;

            document.getElementById('total-b').innerText = `$ ${tB.toLocaleString()}`;

            document.getElementById('grand-total').innerText = `$ ${(tA + tB).toLocaleString()}`;

            document.getElementById('monthly-total').innerText = `$ ${tM.toLocaleString()}`;

            

            document.getElementById('cat-food').innerText = `$ ${cats['食'].toLocaleString()}`;

            document.getElementById('cat-cloth').innerText = `$ ${cats['衣'].toLocaleString()}`;

            document.getElementById('cat-home').innerText = `$ ${cats['住'].toLocaleString()}`;

            document.getElementById('cat-trans').innerText = `$ ${cats['行'].toLocaleString()}`;

            document.getElementById('cat-play').innerText = `$ ${cats['樂'].toLocaleString()}`;

        }


        window.addRecord = async () => {

            const n = document.getElementById('item-name').value.trim();

            const aStr = document.getElementById('item-amount').value.replace(/,/g, '');

            const a = Number(aStr);

            if (!aStr || a <= 0) return;

            await addDoc(collection(db, 'artifacts', finalAppId, 'public', 'data', 'expenses'), { 

                name: n || selectedCategory, 

                amount: a, 

                category: selectedCategory, 

                groupId: currentGroupId, 

                userRole: currentUserRole, 

                createdAt: serverTimestamp() 

            });

            document.getElementById('item-name').value = ''; 

            document.getElementById('item-amount').value = '';

        };


        window.closeDeleteModal = () => document.getElementById('delete-modal').classList.add('hidden');

        document.getElementById('final-delete-btn').onclick = async () => {

            if (pendingDeleteId) {

                await deleteDoc(doc(db, 'artifacts', finalAppId, 'public', 'data', 'expenses', pendingDeleteId));

                window.closeDeleteModal();

            }

        };

    </script>
